My interests lie in commutative algebra, with emphasis on its connections to combinatorics. I mainly work with monomial ideals in the polynomial ring $R = \bbK[X_1,\dots,X_d]$ with $\bbK$ a field and \emph{edge-weighted} graphs $G_\omega$ where $G = (V,E)$ is a finite simple graph with $V = \{v_1,\dots,v_d\}$ and a weight function $\omega: E \to \bbN$ with $\bbN = \{1,2,\dots\}$.

The notion of a simplicial complex can be used to compute irreducible decompositions of square-free monomial ideals. In order to study some kinds of non-square-free monomial ideals, I defined a \emph{generalized $\bbN$-weighted simplicial complex} $\Upsilon$ on $V$, which is a non-empty collection of sets of form $(V',\delta')$ with $V' \subseteq V$ and $\delta': V' \to \bbN$ that is closed under $\preccurlyeq$, where for $(V_1',\delta_1'),(V_2',\delta_2') \in \Upsilon$, write $(V_1',\delta_1') \preccurlyeq (V_2',\delta_2')$ if $V_1' \subseteq V_2'$ and $\delta_1' \leq \delta_2'|_{V_2'}$. Furthermore, to obtain a finite version of generalized weighted simplicial complex, we restrict $\delta'$ such that $m_i \leq \delta'(v_i) \leq M_i$ for $i = 1,\dots,d$.  Then I defined an \emph{generalized $\bbN$-weighted simplicial complex} $\Upsilon^{\bfM}$ with respect to $\bfM$ by
\[\Upsilon^{\bfM} = \{(V',\delta') \in \Upsilon \mid m_i \leq \delta'(v_i) \leq M_i, \fa v_i \in V'\},\] 
where $\bfM$ can be thought of a matrix in $\bbN_{2 \times d}$ whose $i^{\text{th}}$ column is $\begin{bmatrix}m_i \\ M_i\end{bmatrix}$. Associated to $\Upsilon$, there is a natural definition of \emph{weighted Stanley Reisner ideal} $J_{\Upsilon^{\bfM}}$ with respect to $\bfM$, given by
\begin{align*}
    J_{\Upsilon^{\bfM}} &= \Bigl(\underline X^{(V',\delta')} \mathrel{\big |}V' \subseteq V, \ \delta': V' \to \bbN,\ m_i \leq \delta'(v_i) \leq M_i, \fa v_i \in V' \text{ and }(V',\delta') \not\in \Upsilon^{\bfM}\Bigr)R.
\end{align*}

The following definitions will be used in finding irreducible decompositions of $J_{\Upsilon^{\bfM}}$. 

\begin{definition}
    Let $\rho = (V',\delta')$ with $V' \subseteq V$ and $\delta': V' \to \bbN$. Let
    \[W_\rho^{\bfM} = \{v_i \in V' \mid \delta'(v_i) < M_i\},\]
    \[U_\rho^{\bfM} = \{v_i \in V' \mid \delta'(v_i) > m_i\}.\]
    Define $\tau_\rho^{\bfM}: (V \smallsetminus V') \sqcup W_\rho^{\bfM} \to \bbN$ by
    \[\tau_\rho^{\bfM}(v_i) = 
        \left\{\begin{array}{ll}
                m_i & \text{if }v_i \in V \smallsetminus {V'}, \\
                \delta'(v_i) + 1 & \text{if }v_i \in W_\rho^{\bfM},
        \end{array}\right.
    \]
    and $\gamma_\rho^{\bfM}: (V \smallsetminus V') \sqcup U_\rho^{\bfM} \to \bbN$ by
    \[\gamma_\rho^{\bfM}(v_i) = 
        \left\{\begin{array}{ll}
                M_i & \text{if } v_i \in V \smallsetminus {V'}, \\
                \delta'(v_i) - 1 & \text{if }v_i \in U_\rho^{\bfM}.
        \end{array}\right.
    \]
    Let $Q_\rho^{\bfM} \subseteq R$ be the ideal defined by
    \begin{align*}
        Q_\rho^{\bfM} & = (X^{\tau_\rho^{\bfM}(v_i)} \mid v_i \in (V \smallsetminus {V'}) \sqcup W_\rho^{\bfM})R \\
        & = P((V \smallsetminus {V'}) \sqcup W_\rho^{\bfM}, \tau_\rho^{\bfM}).
    \end{align*}
\end{definition}

With previous definitions I obtain an irrredundant irreducible decomposition of $J_{\Upsilon^\bfM}$:

\begin{theorem} \label{JUbfMDecompositionThm}
    The ideal $J_{\Upsilon^\bfM}$ has the following irreducible decomposition 
    \[J_{\Upsilon^\bfM} = \bigcap_{\rho \in \Upsilon^\bfM}Q_\rho^{\bfM} = \bigcap_{\rho \text{ facet}} Q_\rho^{\bfM},\]
    where the first intersection is taken over all faces of $\Upsilon^\bfM$ and the second intersection is taken over all facets of $\Upsilon^\bfM$. The second intersection is irredundant.
\end{theorem}

Just like that we have an equality for edge ideal $I(G)$ of the graph $G$ and Stanley Reisner ideal $J_{\Delta_G}$ of independence complex $\Delta_G$, we also have a weighted version of that with repsect to $\bfM$.

\begin{definition}
    We say that $(V',\delta')$ with $V' \subseteq V$ and $\delta': V' \to \bbN$ is a \emph{weighted independence} set in $G_\omega$ if for any $v_i,v_j \in V'$, either $v_iv_j \not \in E$ or $\min\{\delta'(v_i),\delta'(v_j)\} < \omega(v_iv_j)$. Let $\Upsilon_\omega$ denote the set of weighted independence sets $(V',\delta')$ in $G_\omega$. This is the \emph{weighted independence complex} of $G_\omega$.
\end{definition}

\begin{definition}
    A \emph{weighted vertex cover} of $G_\omega$ with respect to $\bfM$ is a weighted vertex cover $(V',\delta')$ such that $m_i \leq \delta'(v_i) \leq M_i$ for any $v_i \in V'$. Let $\llV_\omega^\bfM$ be the set of weighted vertex cover of $G_\omega$ with respect to $\bfM$.
\end{definition}

\begin{theorem} \label{IGWEJUWMThm}
    If $m_i \leq \min\{\omega(v_iv_j) \mid v_iv_j \in E\}$ and $\max\{\omega(v_iv_j) \mid v_iv_j \in E\} \leq M_i$ for $i = 1,\dots,d$, then
    \[I(G_\omega) = J_{\Upsilon_\omega^\bfM},\]
    where 
    \[I(G_\omega) := (X_i^{\omega(v_iv_j)}X_j^{\omega(v_iv_j)} \mid v_iv_j \in E)R.\]
\end{theorem}

\begin{lemma} \label{mWIIffMWVCLem} 
    Let $m_i \leq \min\{\omega(v_iv_j) \mid v_iv_j \in E\}$ and $\max\{\omega(v_iv_j) \mid v_iv_j \in E\} \leq M_i$ for $i = 1,\dots,d$. Let $\rho = (V',\delta')$ be with $V' \subseteq V$ and $\delta':V' \to \bbN$ such that $m_i \leq \delta'(v_i) \leq M_i$ for any $v_i \in V'$, 
    \begin{enumerate}
        \item \label{mWIIffMWVCLema} 
            $\rho \in \Upsilon_\omega^\bfM$ (is maximal) if and only if $\wp := ((V \smallsetminus {V'}) \sqcup W_\rho^\bfM,\tau_\rho^\bfM) \in \llV_\omega^\bfM$ (is minimal). 
        \item \label{mWIIffMWVCLemb}
            $\rho \in \llV_\omega^\bfM$ is minimal if and only if $\wp := ((V \smallsetminus {V'}) \sqcup U_\rho^\bfM,\gamma_\rho^\bfM) \in \Upsilon_\omega^\bfM$ (is maximal).
    \end{enumerate}
\end{lemma}

\begin{theorem} \label{wEIDecomInWIAndMIThm}
    Let $m_i \leq \min\{\omega(v_iv_j) \mid v_iv_j \in E\}$ and $\max\{\omega(v_iv_j) \mid v_iv_j \in E\} \leq M_i$ for $i = 1,\dots,d$. Then the ideal $J_{\Upsilon_\omega^\bfM}$ has the following irreducible decompositions
    \[I(G_\omega) =  \bigcap_{\rho \in \Upsilon_\omega^\bfM} Q_\rho^\bfM = \bigcap_{\text{maximal } \rho \in \Upsilon_\omega^\bfM} Q_\rho^\bfM.\]
    The second intersection is irredundant.
\end{theorem}

\begin{definition}
    Define the \emph{Alexander dual} of $\Upsilon^\bfM$ by
    \[(\Upsilon^\bfM)^\vee = \{\rho := (V',\delta') \text{ with }V' \subseteq V, \ \delta': V' \to \bbN,\text{ and }  m_i \leq \delta'(v_i) \mid ((V \smallsetminus {V'}) \sqcup W_\rho^\bfM,\varphi_\rho^\bfM) \not\in \Upsilon^\bfM\}.\]
    Then $(\Upsilon^\bfM)^\vee$ is a weighted simplicial complex with respect to $\bfM$ closed under ``$\preccurlyeq$''. 
\end{definition}

\begin{definition}
    Write $I = \bigcap_{i=1}^n P(V_i',\delta_i')$ as a irredundant irreducible decomposition. Define the \emph{Alexander dual} of $I$ by
    \[I^\vee = (\underline X^{(V_i',\delta_i'')} \mid i = 1,\dots,n),\]
    where for $i = 1,\dots,n$: $\delta_i''(v_j) = M_j + m_i - \delta_i'(v_j)$ for any $v_j \in V_i'$.
\end{definition}

If $m_i = 1$ for $i = 1,\dots,d$, then the definition is consistent with the one from \emph{Alexander Duality for Monomial Ideals and Their Resolutions} by Ezra Miller.

\begin{theorem}
    \[(J_{\Upsilon^\bfM})^\vee = J_{(\Upsilon^\bfM)^\vee}.\]
\end{theorem}

\begin{center}
    \textsl{\large 1. Cohen-Macaulay type of weighted edge ideals}
\end{center}

One of my main research goals is to compute combinatorially the (finite) dimension of $\bbK$-vector space $\operatorname{Ext}_R^{n}(\bbK,R/I)$ with $n := \operatorname{depth}_R(R/I)$, where $I$ comes from edge-weighted graphs as a monomial ideal in $R$. Regard $R/I$ as a \emph{graded $R$-module} with natural grading, then $R/I$ behaves like a \emph{local ring} with the unique \emph{maximal homogeneous ideal} $\ffX/I$ where $\ffX = (X_1,\dots,X_d)R$. From now on all algebraic objects and morphisms will be homogeneous or graded by default. In this setting the $R$-module $\operatorname{Ext}_R^n(\bbK,R/I)$ is also called the \emph{type} of $R/I$, denoted by
    \[\operatorname{r}_R(R/I) = \dim_\bbK(\operatorname{Ext}_R^n(\bbK,R/I)).\]
    Since the notation of Cohen-Macaulay ring is a workhorse of commutative algebra and have been shown over and over again in the literature to be extremely nice. I put my interests in computing $\operatorname{r}_R(R/I)$ when $R/I$ is a Cohen-Macaulay $R$-module and again $I \subseteq R$ is a monomial ideal contructed combinatorially from edge-weighted graphs. 

    If $R/I(G_\omega)$ is Cohen-Macaulay and $\dim(R/I(G_\omega)) = 0$, then $\operatorname{depth}(R/I(G_\omega)) = 0$ and so
    \[\operatorname{type}(R/I(G_\omega)) = \dim_\bbK(\operatorname{Ext}_R^0(\bbK,R/I(G_\omega))) = \dim_\bbK(\operatorname{Hom}_R(\bbK,R/I(G_\omega))) = t,\]
    where $t$ is the number of \emp{parametric ideals} occuring in an irredundant \emhp{parametric decomposition} of $I(G_\omega)$, which is also the the number of irreducible ideals occuring in the irredundant irreducible decomposition of $I(G_\omega)$.


\begin{theorem}\label{computeTypeOfWRPathOfWRPathSuspension}
    Let $(\Sigma_rG)_\lambda$ be a weighted $r$-path suspension of $G_\omega$ such that $\lambda(v_iv_j) \leq \lambda(v_i,v_{i,1})$ and $\lambda(v_iv_j) \leq \lambda(v_j,v_{j,1})$ for all edges $v_iv_j \in E$. 
    \[\operatorname{r}_{R'}\biggl(\frac{R'}{I_r((\Sigma_r G)_\lambda)}\biggr) = \sharp\, \{\text{$\mathcal p$-minimal weighted $r$-path vertex covers of $(\Sigma_{r-1}G)_{\lambda'}$}\}, \ \lambda' = \lambda|_{\Sigma_{r-1}G}.\]
\end{theorem}

Set $r = 1$ in Theorem~\ref{computeTypeOfWRPathOfWRPathSuspension}, then a $\mathcal p$-minimal weighted 1-path vertex cover of $(\Sigma_1G)_\lambda$ becomes a minimal weighted vertex cover of $(\Sigma G)_\lambda$ and $\lambda' = \lambda|_{\Sigma_0G} = \lambda|_G = \omega$, so we get the following corollary.

\begin{corollary} \label{weightedCMTypeEqualToMinWVCTheorem}
    Let $(\Sigma G)_\lambda$ be a weighted suspension of $G_\omega$ such that $\lambda(v_iv_j) \leq \lambda(v_iw_i)$ and $\lambda(v_iv_j) \leq \lambda(w_jv_j)$ for each $v_iv_j \in E$. Then
    \[\operatorname{r}_{R'}\biggl(\frac{R'}{I((\Sigma G)_\lambda)}\biggr) = \sharp\, \{\text{minimal weighted vertex covers of }G_\omega\}.\]
\end{corollary}

\begin{center}
    \textsl{\large 2. Cohen-Macaulay edge-weighted chordal graphs classifying}
\end{center}

Finally, I classified all Cohen-Macaulay edge-weighted chordal graphs following the classification of all Cohen-Macaulay chordal graphs by J\"urgen. The characterization is purely graph-theoretical, and it turns out that for weighted chordal graphs the Cohen-Macaulay property is independent of the field $\bbK$.

\begin{theorem} \label{CMUnmixedEquivalenceThm}
    Let $G_\lambda$ be a edge-weighted chordal graph. Let $F_1,\dots,F_m$ be the facets of $\Delta(G)$ which admit a free vertex satisfying that for $i = 1,\dots,m$: there doesn't exists a rooted spanning forest $\ffF$ of $G[F_i]$ in which each component $T$ has a nonfree vertex $\nu_1^T$ as the root such that there is a nonfree vertex $\nu_0^T$ in $V \smallsetminus F_i$ with $\nu_0^T\nu_1^T \in E$ which satisfies that for any path $\nu_1^Tv_2 \cdots v_k$ in $T$ with $v_k$ a $T$-leaf if $k \geq 2$, then $\lambda(\nu_0^T\nu_1^T) > \lambda(\nu_1^Tv_2) > \lambda(v_2v_3) > \cdots > \lambda(v_{k-1}v_k)$, and which satisfies that for any component(s) $S,T$ of $\ffF$ if $u$ is an $S$-inner vertex and $v$ is a $T$-inner vertex with $u \neq v$, then 
    \[\lambda(uv) > \min\{\max\{\lambda(uw) \mid \text{$u$ is the $S$-parent of $w$}\},\max\{\lambda(vx) \mid \text{$v$ is the $T$-parent of $x$}\}\};\]
    if $u$ is an $S$-inner vertex such that $u \nu_0^T \in E$, then
    \begin{align*}
        \lambda(u\nu_0^T) &> \min\{\max\{\lambda(uw) \mid \text{$u$ is the $S$-parent of $w$}\}, \\
        &\ \ \ \ \ \ \ \ \ \, \max\{\lambda(\nu_0^T\nu_1^Y) \mid \text{$Y$ is a component of $\ffF$ such that $\nu_0^T = \nu_0^Y$}\}\};
    \end{align*}
    and if $\nu_0^S\nu_0^T \in E$, then 
    \begin{align*}
        \lambda(\nu_0^S\nu_0^T) &> \min\{\max\{\lambda(\nu_0^S\nu_1^Y) \mid \text{$Y$ is a component of $\ffF$ such that $\nu_0^S = \nu_0^Y$}\}, \\
        &\ \ \ \ \ \ \ \ \ \, \max\{\lambda(\nu_0^T\nu_1^Z) \mid \text{$Z$ is a component of $\ffF$ such that $\nu_0^T = \nu_0^Z$}\}\}.
    \end{align*}
    Then the following conditions are equivalent.
    \begin{enumerate} 
        \item $G_\lambda$ is Cohen-Macaulay;
        \item $G_\lambda$ is Cohen-Macaulay over $\bbK$;
        \item $G_\lambda$ is unmixed;
        \item $[n]$ is the disjoint union of $F_1,\dots,F_m$.
    \end{enumerate}
\end{theorem}

\begin{theorem} \label{CMUnmixedEquivalenceNecessaryThm}
    Let $G_\lambda$ be a edge-weighted chordal graph. Let $F_1,\dots,F_m$ be the facets of $\Delta(G)$ which admit a free vertex satisfying that $[n]$ is the disjoint union of $F_1,\dots,F_m$. If $G_\lambda$ is Cohen-Macaulay, then there doesn't exists a rooted spanning forest $\ffF$ of $G[F_i]$ in which each component $T$ has a nonfree vertex $\nu_1^T$ as the root such that there is a nonfree vertex $\nu_0^T$ in $V \smallsetminus F_i$ with $\nu_0^T\nu_1^T \in E$ which satisfies that for any path $\nu_1^Tv_2 \cdots v_k$ in $T$ with $v_k$ a $T$-leaf if $k \geq 2$, then $\lambda(\nu_0^T\nu_1^T) > \lambda(\nu_1^Tv_2) > \lambda(v_2v_3) > \cdots > \lambda(v_{k-1}v_k)$, and which satisfies that for any component(s) $S,T$ of $\ffF$ if $u$ is an $S$-inner vertex and $v$ is a $T$-inner vertex with $u \neq v$, then 
    \[\lambda(uv) > \min\{\max\{\lambda(uw) \mid \text{$u$ is the $S$-parent of $w$}\},\max\{\lambda(vx) \mid \text{$v$ is the $T$-parent of $x$}\}\};\]
    if $u$ is an $S$-inner vertex such that $u \nu_0^T \in E$, then
    \begin{align*}
        \lambda(u\nu_0^T) &> \min\{\max\{\lambda(uw) \mid \text{$u$ is the $S$-parent of $w$}\}, \\
        &\ \ \ \ \ \ \ \ \ \, \max\{\lambda(\nu_0^T\nu_1^Y) \mid \text{$Y$ is a component of $\ffF$ such that $\nu_0^T = \nu_0^Y$}\}\};
    \end{align*}
    and if $\nu_0^S\nu_0^T \in E$, then 
    \begin{align*}
        \lambda(\nu_0^S\nu_0^T) &> \min\{\max\{\lambda(\nu_0^S\nu_1^Y) \mid \text{$Y$ is a component of $\ffF$ such that $\nu_0^S = \nu_0^Y$}\}, \\
        &\ \ \ \ \ \ \ \ \ \, \max\{\lambda(\nu_0^T\nu_1^Z) \mid \text{$Z$ is a component of $\ffF$ such that $\nu_0^T = \nu_0^Z$}\}\}.
    \end{align*}
\end{theorem}
